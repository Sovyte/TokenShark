Integration

Three lines of code.
Zero configuration.

# 1. Install
pip install tokenshark

# 2. Import & wrap
from tokenshark import monitor
monitor.init(api_key="ts_your_key_here")

# 3. Use normally — we handle the rest
import openai
response = openai.chat.completions.create(
    model="gpt-4o",
    messages=[{"role": "user", "content": "Hello"}]
)
# → Automatically tracked: cost, latency, tokens, quality
// 1. Install
npm install tokenshark

// 2. Import & wrap
const { monitor } = require('tokenshark');
monitor.init({ apiKey: 'ts_your_key_here' });

// 3. Use normally — we handle the rest
const openai = require('openai');
const response = await openai.chat.completions.create({
    model: 'gpt-4o',
    messages: [{ role: 'user', content: 'Hello' }]
});
// → Automatically tracked: cost, latency, tokens, quality
# Send event directly via API
curl -X POST https://api.tokenshark.io/v1/ingest \
  -H "X-API-Key: ts_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{
    "provider": "openai",
    "model": "gpt-4o",
    "prompt_tokens": 150,
    "completion_tokens": 420,
    "cost_usd": 0.0124,
    "latency_ms": 234,
    "feature_tag": "chat"
  }'
